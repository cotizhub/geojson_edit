// Configurar eventos de teclado mejorados function setupKeyboardEvents() { document.addEventListener('keydown', function(e) { // Si estamos en un input, no procesar atajos de teclado if (document.activeElement.tagName === 'INPUT') { if (e.key === 'Enter') { e.preventDefault(); document.activeElement.blur(); if (activeCell) { moveActiveCell('down', false); } } else if (e.key === 'Tab') { e.preventDefault(); document.activeElement.blur(); if (activeCell) { moveActiveCell(e.shiftKey ? 'left' : 'right', false); } } else if (e.key === 'Escape') { e.preventDefault(); document.activeElement.blur(); } return; } // Ctrl+A para seleccionar todo if (e.ctrlKey && e.key === 'a') { e.preventDefault(); selectAll(); return; } // Ctrl+Z para deshacer if (e.ctrlKey && e.key === 'z') { e.preventDefault(); undoAction(); return; } // Ctrl+Y para rehacer if (e.ctrlKey && e.key === 'y') { e.preventDefault(); redoAction(); return; } // Ctrl+O para abrir archivo if (e.ctrlKey && e.key === 'o') { e.preventDefault(); document.getElementById('fileInput').click(); return; } // Ctrl+S para guardar if (e.ctrlKey && e.key === 's') { e.preventDefault(); saveData(); return; } // Ctrl+D para descargar if (e.ctrlKey && e.key === 'd') { e.preventDefault(); downloadFile(); return; } // Ctrl+F para buscar if (e.ctrlKey && e.key === 'f') { e.preventDefault(); searchData(); return; } // F3 para buscar siguiente if (e.key === 'F3') { e.preventDefault(); if (e.shiftKey) { findPrevMatch(); // Shift+F3 para buscar anterior } else { findNextMatch(); // F3 para buscar siguiente } return; } // Delete para eliminar selección if (e.key === 'Delete') { e.preventDefault(); deleteSelected(); return; } // Ctrl+C para copiar if (e.ctrlKey && e.key === 'c') { e.preventDefault(); copySelection(); return; } // Ctrl+V para pegar if (e.ctrlKey && e.key === 'v') { e.preventDefault(); pasteSelection(); return; } // Navegación con flechas (con o sin Shift) if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key)) { e.preventDefault(); navigateWithArrows(e.key, e.shiftKey); return; } // Enter para editar celda activa if (e.key === 'Enter' && activeCell) { e.preventDefault(); startEditingActiveCell(); return; } // Escape para limpiar selección if (e.key === 'Escape') { e.preventDefault(); clearSelection(); return; } }); } // Función para seleccionar todas las celdas function selectAll() { selectedCells.clear(); // Seleccionar todas las celdas, incluyendo encabezados for (let row = -1; row < geoData.features.length; row++) { for (let col = 0; col < currentColumns.length; col++) { selectedCells.add(`${row},${col}`); } } // Establecer la celda activa en la primera celda de datos if (geoData.features.length > 0) { setActiveCell(0, 0, false); // No limpiar selección } // Marcar la esquina superior izquierda como seleccionada document.querySelector('.corner-header-cell').classList.add('all-selected'); highlightSelectedCells(); updateSelectionInfo(); updateStatus("Todas las celdas seleccionadas"); } // Navegación con flechas mejorada function navigateWithArrows(direction, withShift) { if (!activeCell) { // Si no hay celda activa y hay datos, seleccionar la primera celda if (geoData.features.length > 0) { setActiveCell(0, 0); } return; } switch (direction) { case 'ArrowUp': moveActiveCell('up', withShift); break; case 'ArrowDown': moveActiveCell('down', withShift); break; case 'ArrowLeft': moveActiveCell('left', withShift); break; case 'ArrowRight': moveActiveCell('right', withShift); break; } } // Mover celda activa con soporte para selección con Shift function moveActiveCell(direction, withShift) { if (!activeCell) return; let newRow = activeCell.row; let newCol = activeCell.col; // Guardar la posición anterior para la selección con Shift const oldRow = newRow; const oldCol = newCol; switch (direction) { case 'up': newRow = Math.max(-1, newRow - 1); break; case 'down': newRow = Math.min(geoData.features.length - 1, newRow + 1); break; case 'left': newCol = Math.max(0, newCol - 1); break; case 'right': newCol = Math.min(currentColumns.length - 1, newCol + 1); break; } // Si estamos usando Shift para seleccionar un rango if (withShift) { // Si no hay selección previa, iniciar desde la celda activa if (!selectionStartCell) { selectionStartCell = { row: oldRow, col: oldCol }; } selectionEndCell = { row: newRow, col: newCol }; // Actualizar la selección basada en el rango updateRangeSelection(); // Actualizar la celda activa sin limpiar la selección document.querySelectorAll('.cell-active').forEach(cell => cell.classList.remove('cell-active')); activeCell = { row: newRow, col: newCol }; const cell = document.querySelector(`.sheet-table td[data-row="${newRow}"][data-col="${newCol}"]`); if (cell) { cell.classList.add('cell-active'); cell.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'nearest' }); } } else { // Comportamiento normal sin Shift setActiveCell(newRow, newCol); // Resetear los puntos de inicio y fin de selección selectionStartCell = null; selectionEndCell = null; } } // Actualizar selección basada en rango function updateRangeSelection() { if (!selectionStartCell || !selectionEndCell) return; const maxCol = Math.max(selectionStartCell.col, selectionEndCell.col); for (let r = minRow; r <= maxRow; r++) { for (let c = minCol; c <= maxCol; c++) { selectedCells.add(`${r},${c}`); } } highlightSelectedCells(); updateSelectionInfo(); } // Establecer celda activa con opción para no limpiar selección function setActiveCell(row, col, clearSelection = true) { document.querySelectorAll('.cell-active').forEach(cell => cell.classList.remove('cell-active')); } if (clearSelection) { selectedCells.clear(); selectedCells.add(`${row},${col}`); highlightSelectedCells(); updateSelectionInfo(); } } function startEditingActiveCell() { if (!activeCell) return; const cell = document.querySelector(`.sheet-table td[data-row="${activeCell.row}"][data-col="${activeCell.col}"] input`); if (cell) { cell.focus(); cell.select(); } } function clearSelection() { selectedCells.clear(); activeCell = null; selectionStartCell = null; selectionEndCell = null; document.querySelectorAll('.cell-active, .cell-selected').forEach(cell => { cell.classList.remove('cell-active', 'cell-selected'); }); document.querySelectorAll('.row-number-cell').forEach(cell => cell.classList.remove('row-selected')); document.querySelectorAll('.sheet-table th').forEach(cell => cell.classList.remove('column-selected')); } // Manejar clicks en encabezados de columna if (e.target.tagName === 'TH' && !e.target.classList.contains('corner-header-cell')) { const colIndex = parseInt(e.target.dataset.index); if (!isNaN(colIndex)) { if (e.shiftKey && activeCell) { // Seleccionar rango de columnas selectColumnRange(activeCell.col, colIndex); if (cell) { const row = parseInt(cell.dataset.row); const col = parseInt(cell.dataset.col); if (e.shiftKey && activeCell) { // Seleccionar rango de celdas selectionStartCell = { row: activeCell.row, col: activeCell.col }; selectionEndCell = { row, col }; updateRangeSelection(); setActiveCell(row, col, false); // No limpiar selección } else if (!e.ctrlKey) { setActiveCell(row, col); // Limpiar otras selecciones visuales document.querySelectorAll('.row-number-cell').forEach(cell => cell.classList.remove('row-selected')); document.querySelectorAll('.sheet-table th').forEach(cell => cell.classList.remove('column-selected')); document.querySelector('.corner-header-cell').classList.remove('all-selected'); } if (e.shiftKey && activeCell) { // Seleccionar rango de celdas selectionStartCell = { row: activeCell.row, col: activeCell.col }; selectionEndCell = { row, col }; updateRangeSelection(); setActiveCell(row, col, false); // No limpiar selección } else if (e.ctrlKey) { // Añadir/quitar celda a la selección toggleCellSelection(row, col); if (cell) { const input = cell.querySelector('input'); if (input) { input.focus(); input.select(); } } }); } // Seleccionar una columna function selectColumn(colIndex) { selectedCells.clear(); // Seleccionar todas las filas de esta columna (incluyendo la fila de nombres) for (let rowIndex = -1; rowIndex < geoData.features.length; rowIndex++) { selectedCells.add(`${rowIndex},${colIndex}`); } // Marcar visualmente la columna como seleccionada document.querySelectorAll('.sheet-table th').forEach(cell => cell.classList.remove('column-selected')); } // Establecer la celda activa en la primera celda de la columna if (geoData.features.length > 0) { setActiveCell(0, colIndex, false); // No limpiar selección } highlightSelectedCells(); updateSelectionInfo(); const excelColName = getExcelColumnName(colIndex); const propertyName = columnNameMap[excelColName]; updateStatus(`Columna ${excelColName} (${propertyName}) seleccionada`); } // Seleccionar un rango de columnas function selectColumnRange(startColIndex, endColIndex) { selectedCells.clear(); const minCol = Math.min(startColIndex, endColIndex); const maxCol = Math.max(startColIndex, endColIndex); // Seleccionar todas las filas de estas columnas for (let rowIndex = -1; rowIndex < geoData.features.length; rowIndex++) { for (let colIndex = minCol; colIndex <= maxCol; colIndex++) { selectedCells.add(`${rowIndex},${colIndex}`); } } // Marcar visualmente las columnas como seleccionadas document.querySelectorAll('.sheet-table th').forEach(cell => cell.classList.remove('column-selected')); if (colHeader) { colHeader.classList.add('column-selected'); } } // Establecer la celda activa en la primera celda de la columna if (geoData.features.length > 0) { setActiveCell(0, colIndex, false); // No limpiar selección } highlightSelectedCells(); updateSelectionInfo(); const excelColName = getExcelColumnName(colIndex); const propertyName = columnNameMap[excelColName]; updateStatus(`Columna ${excelColName} (${propertyName}) ${isSelected ? 'deseleccionada' : 'añadida a la selección'}`); } // Seleccionar una fila function selectRow(rowIndex) { selectedCells.clear(); // Seleccionar todas las columnas de esta fila for (let colIndex = 0; colIndex < currentColumns.length; rowIndex++) { for (let colIndex = 0; colIndex < currentColumns.length; colIndex++) { selectedCells.add(`${rowIndex},${colIndex}`); } } // Marcar visualmente las filas como seleccionadas document.querySelectorAll('.row-number-cell').forEach(cell => cell.classList.remove('row-selected')); } }); // Añadir fila de nombres de propiedades const propertyNameRow = document.createElement('tr'); propertyNameRow.className = 'property-name-row'; // Añadir celda de número de fila const rowNumberCell = document.createElement('td'); rowNumberCell.className = 'row-number-cell'; rowNumberCell.textContent = '1'; rowNumberCell.dataset.index = -1; propertyNameRow.appendChild(rowNumberCell); // Añadir celdas de nombres de propiedades currentColumns.forEach((col, colIndex) => { const td = document.createElement('td'); delete feature.properties[oldName]; } }); updateStatus(`Columna ${excelColName} renombrada de "${oldName}" a "${newName}"`); } else { // Restaurar nombre original si el nuevo no es válido this.value = oldName; if (newName === oldName) { // No hacer nada, es el mismo nombre } else if (!newName) { updateStatus(`Error: El nombre de la columna no puede estar vacío`, true); } else if (currentColumns.includes(newName)) { updateStatus(`Error: Ya existe una columna con el nombre "${newName}"`, true); } } }); td.appendChild(input); propertyNameRow.appendChild(td); }); tableBody.appendChild(propertyNameRow); // Añadir filas de datos geoData.features.forEach((feature, rowIndex) => { const tr = document.createElement('tr'); // Añadir celda de número de fila const rowNumberCell = document.createElement('td'); rowNumberCell.className = 'row-number-cell'; rowNumberCell.textContent = rowIndex + 2; // +2 porque la primera fila es de nombres y empezamos en 1 rowNumberCell.dataset.index = rowIndex; tr.appendChild(rowNumberCell); // Añadir celdas de datos currentColumns.forEach((col, colIndex) => { const td = document.createElement('td'); td.dataset.row = rowIndex; td.dataset.col = colIndex; const input = document.createElement('input'); } else { input.value = ''; } // Manejar cambios en las coordenadas input.addEventListener('change', function() { saveToHistory(); const coords = this.value.split(',').map(c => parseFloat(c.trim())); if (coords.length >= 2 && !isNaN(coords[0]) && !isNaN(coords[1])) { if (!feature.geometry) { feature.geometry = { type: 'Point' }; Mostrando todas las filas."); } // Actualizar indicador visual de filtro function updateFilterIndicator() { // Limpiar indicadores previos document.querySelectorAll('.filter-indicator').forEach(el => el.remove()); if (filterActive && currentFilter) { const th = document.querySelector(`th[data-index="${currentFilter.colIndex}"]`); if (th) { const indicator = document.createElement('span'); indicator.className = 'filter-indicator'; indicator.textContent = ' 🔍'; indicator.title = 'Esta columna tiene un filtro activo'; th.appendChild(indicator); } } } // Función para buscar en los datos function searchData() { // Crear un diálogo modal para la búsqueda const modal = document.createElement('div'); modal.className = 'modal'; // Crear contenido del modal modal.innerHTML = `

document.body.removeChild(modal); } } }); } // Buscar coincidencias en los datos function findMatches() { if (!searchTerm) return []; const matches = []; const term = searchTerm.toLowerCase(); // Buscar en todas las celdas geoData.features.forEach((feature, rowIndex) => { currentColumns.forEach((col, colIndex) => { let cellValue; if (col === 'geometry') { cellValue = feature.geometry?.coordinates?.join(', ') || ''; } else { cellValue = feature.properties?.[col] || ''; } // Convertir a string para búsqueda cellValue = String(cellValue).toLowerCase(); if (cellValue.includes(term)) { matches.push({ row: rowIndex, col: colIndex }); } }); }); return matches; } // Encontrar la siguiente coincidencia function findNextMatch() { // Buscar todas las coincidencias searchResults = findMatches(); updateFindOptions(); if (searchResults.length === 0) { updateStatus(`No se encontraron coincidencias para "${searchTerm}"`, true); return; } // Avanzar al siguiente resultado currentSearchIndex = (currentSearchIndex + 1) % searchResults.length; } // Función para validar la estructura GeoJSON function validateGeoJSON(silent = false) { const errors = []; // Verificar tipo de colección if (geoData.type !== 'FeatureCollection') { errors.push('El tipo debe ser "FeatureCollection"'); } // Verificar que features sea un array if (!Array.isArray(geoData.features)) { errors.push('La propiedad "features" debe ser un array'); } else { // Verificar cada feature geoData.features.forEach((feature, index) => { if (feature.type !== 'Feature') { errors.push(`Feature #${index + 1}: El tipo debe ser "Feature"`); } if (!feature.geometry) { errors.push(`Feature #${index + 1}: Falta la propiedad "geometry"`); } else { if (!feature.geometry.type) { errors.push(`Feature #${index + 1}: Falta el tipo de geometría`); } if (!feature.geometry.coordinates) { errors.push(`Feature #${index + 1}: Faltan las coordenadas`); } else { // Verificar coordenadas según el tipo de geometría switch (feature.geometry.type) { case 'Point': if (!Array.isArray(feature.geometry.coordinates) || feature.geometry.coordinates.length !== 2) { errors.push(`Feature #${index + 1}: Las coordenadas de un Point deben ser un array de 2 números`); } break; case 'LineString': if (!Array.isArray(feature.geometry.coordinates) || feature.geometry.coordinates.length < 2) { errors.push(`Feature #${index + 1}: Las coordenadas de un LineString deben ser un array de al menos 2 puntos`); } break; case 'Polygon': if (!Array.isArray(feature.geometry.coordinates) || feature.geometry.coordinates.length < 1) { errors.push(`Feature #${index + 1}: Las coordenadas de un Polygon deben ser un array de anillos`); } break; } } } if (!feature.properties) { errors.push(`Feature #${index + 1}: Falta la propiedad "properties"`); } }); } // Mostrar resultados si no es silencioso if (!silent) { if (errors.length === 0) { updateStatus("La estructura GeoJSON es válida"); } else { // Crear un diálogo modal para mostrar los errores const modal = document.createElement('div'); modal.className = 'modal'; // Crear contenido del modal modal.innerHTML = `

Errores de validación GeoJSON

    ${errors.map(error => `
  • ${error}
    • `).join('')}
`; document.body.appendChild(modal); // Manejar el botón de cerrar document.getElementById('close-validation').addEventListener('click', function() { document.body.removeChild(modal); }); updateStatus(`Se encontraron ${errors.length} errores en la estructura GeoJSON`, true); } } return errors.length === 0; }
